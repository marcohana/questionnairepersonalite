// --- CALCUL DISC ---

function calculateDISC() {

    let values = [];

    for (let i = 0; i < 28; i++) {

        const chosen = document.querySelector(`input[name="q${i}"]:checked`);

        if (!chosen) {

            alert("Veuillez répondre à toutes les questions.");

            return;

        }

        values.push(parseInt(chosen.value));

    }


    let D = sum(values.slice(0,7));

    let I = sum(values.slice(7,14));

    let S = sum(values.slice(14,21));

    let C = sum(values.slice(21,28));


    const scores = { "Dominance": D, "Influence": I, "Stabilité": S, "Conformité": C };

    const profile = Object.keys(scores).reduce((a,b) => scores[a] > scores[b] ? a : b);


    const colorClass = {

        "Dominance": "profile-D",

        "Influence": "profile-I",

        "Stabilité": "profile-S",

        "Conformité": "profile-C"

    }[profile];


    document.getElementById("result").innerHTML = `

        <div class="result">

            <h2>Vos scores DISC</h2>

            <p><strong>Dominance :</strong> ${D}</p>

            <p><strong>Influence :</strong> ${I}</p>

            <p><strong>Stabilité :</strong> ${S}</p>

            <p><strong>Conformité :</strong> ${C}</p>


            <div class="profile-box ${colorClass}">

                Profil dominant : ${profile}

            </div>


            <p style="margin-top:15px;font-size:16px;">${profiles[profile]}</p>

        </div>

    `;

}
